// BLOCK DEFINITION

Blockly.defineBlocksWithJsonArray([
  {
    "type": "open_gate_with_speed",
    "message0": "Open Gate With Speed %1",
    "args0": [
      {
        "type": "field_number",
        "name": "SPEED",
        "value": 5,
        "min": 1,
        "max": 10
      }
    ],
    "previousStatement": null,
    "nextStatement": null,
    "colour": 210,
    "tooltip": "Opens the gate using servo motor with selected speed",
    "helpUrl": ""
  }
]);


// JAVASCRIPT GENERATOR

javascript.javascriptGenerator.forBlock['open_gate_with_speed'] = function(block) {

  // Read speed value
  var speed = block.getFieldValue('SPEED');

  // Convert speed to servo angle
  var angle = speed * 9;

  // Generate servo code
  var code = `
servo.write(${angle});
delay(500);
`;

  return code;
};
